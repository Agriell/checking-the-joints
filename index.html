        // Показать вопрос

        function showQuestion(index) {

            const q = questions[index];

            let html = `<div class="question-title">${q.text}</div>`;

            

            if (q.hint) {

                html += `<div class="question-hint">${q.hint}</div>`;

            }

            

            html += `<div class="options-group" id="options-group">`;

            

            q.options.forEach((opt, optIndex) => {

                const optionId = `q${index}_opt${optIndex}`;

                const checked = answers[`q${index}`] && 

                    (Array.isArray(answers[`q${index}`]) 

                        ? answers[`q${index}`].includes(opt.id)

                        : answers[`q${index}`] === opt.id);

                

                if (q.type === 'multiple') {

                    html += `

                        <div class="option-item">

                            <input type="checkbox" id="${optionId}" value="${opt.id}" ${checked ? 'checked' : ''}>

                            <label for="${optionId}">${opt.emoji ? opt.emoji + ' ' : ''}${opt.label}</label>

                        </div>

                    `;

                } else {

                    html += `

                        <div class="option-item">

                            <input type="radio" name="q${index}" id="${optionId}" value="${opt.id}" ${checked ? 'checked' : ''}>

                            <label for="${optionId}">${opt.emoji ? opt.emoji + ' ' : ''}${opt.label}</label>

                        </div>

                    `;

                }

            });

            

            html += `</div>`;

            questionCard.innerHTML = html;

            

            // Обновление счетчика

            const totalQuestions = questions.length;

            questionCounter.innerText = `Вопрос ${index + 1} из ${totalQuestions}`;

            const percent = ((index + 1) / totalQuestions) * 100;

            progressBar.style.width = `${percent}%`;

            progressPercent.innerText = `${Math.round(percent)}%`;

            

            // Кнопка "Назад"

            if (index === 0) {

                prevBtn.style.display = 'none';

            } else {

                prevBtn.style.display = 'block';

            }

            

            // Текст кнопки "Далее"

            if (index === totalQuestions - 1) {

                nextBtn.innerText = 'Показать результат';

            } else {

                nextBtn.innerText = 'Далее';

            }

        }


        // Сохранить ответы на текущий вопрос

        function saveCurrentAnswer() {

            const q = questions[currentQuestionIndex];

            const optionsGroup = document.getElementById('options-group');

            

            if (q.type === 'multiple') {

                const checked = Array.from(optionsGroup.querySelectorAll('input[type="checkbox"]:checked'))

                    .map(input => input.value);

                answers[`q${currentQuestionIndex}`] = checked;

            } else {

                const checked = optionsGroup.querySelector('input[type="radio"]:checked');

                if (checked) {

                    answers[`q${currentQuestionIndex}`] = checked.value;

                }

            }

        }


        // Проверить, отвечен ли текущий вопрос

        function isCurrentQuestionAnswered() {

            const q = questions[currentQuestionIndex];

            const answer = answers[`q${currentQuestionIndex}`];

            

            if (!answer) return false;

            

            if (q.type === 'multiple') {

                return Array.isArray(answer) && answer.length > 0;

            } else {

                return answer !== undefined && answer !== '';

            }

        }


        // Подсчет баллов

        function calculateScore() {

            let total = 0;

            

            // Вопрос 2 (q1 в массиве, индекс 1)

            if (answers['q1']) {

                const q2opt = questions[1].options.find(opt => opt.id === answers['q1']);

                if (q2opt) total += q2opt.points;

            }

            

            // Вопрос 3 (индекс 2)

            if (answers['q2']) {

                const q3opt = questions[2].options.find(opt => opt.id === answers['q2']);

                if (q3opt) total += q3opt.points;

            }

            

            // Вопрос 4 (индекс 3) — множественный

            if (answers['q3'] && Array.isArray(answers['q3'])) {

                answers['q3'].forEach(id => {

                    const opt = questions[3].options.find(o => o.id === id);

                    if (opt) total += opt.points;

                });

            }

            

            // Вопрос 5 (индекс 4)

            if (answers['q4']) {

                const q5opt = questions[4].options.find(opt => opt.id === answers['q4']);

                if (q5opt) total += q5opt.points;

            }

            

            // Бонус за вопрос 6 (индекс 5) — 3+ метода

            if (answers['q5'] && Array.isArray(answers['q5'])) {

                const selected = answers['q5'].filter(id => id !== 'nothing');

                if (selected.length >= 3) {

                    total += 1;

                }

            }

            

            // Вопрос 7 (индекс 6) — только если есть ответ

            if (answers['q6']) {

                const q7opt = questions[6].options.find(opt => opt.id === answers['q6']);

                if (q7opt) total += q7opt.points;

            }

            

            return total;

        }


        // Получить зону из ответа на вопрос 1

        function getZone() {

            return answers['q0'] || 'knee'; // По умолчанию колено, если нет ответа

        }

            if (score <= 3) return 'green';

            if (score <= 6) return 'yellow';

            return 'red';
